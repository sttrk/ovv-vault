
scope_out:
  en: scope_out
  ja: スコープ外

  engineer_view:
    meaning: 実装対象外として明示的に除外された範囲
    implication: この項目に関わる実装変更は仕様逸脱となる
    reversible: false

  pmo_view:
    meaning: 今回は「やらない」と合意された範囲
    implication: 実施する場合は変更管理（CDC）が必要
    reversible: true

  meta:
    responsibility:
      engineer: 実装しないことを保証
      pmo: 合意内容を維持・説明する責任
    collision_risk: high
    notes:
      - エンジニアは「実装境界」として扱う
      - PMOは「合意境界」として扱う
